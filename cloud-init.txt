packages:
  - moby-cli
  - moby-engine
runcmd:
  - |
      set -x
      (
        # Wait for docker daemon to start
        while [ $(ps -ef | grep -v grep | grep docker | wc -l) -le 0 ]; do 
          sleep 3
        done
	
	apt install aziot-identity-service=1.2.0~rc4-1
	apt install aziot-edge=1.2.0~rc4-1 

	cat <<EOF >/etc/aziot/config.toml
	[provisioning]
	source = "manual"
	connection_string = "variables('dcs')"
	 
	[agent]
	name = "edgeAgent"
	type = "docker"
	 
	[agent.config]
	image = "mcr.microsoft.com/azureiotedge-agent:1.1"
	 
	[connect]
	workload_uri = "unix:///var/run/iotedge/workload.sock"
	management_uri = "unix:///var/run/iotedge/mgmt.sock"
	 
	[listen]
	workload_uri = "fd://aziot-edged.workload.socket"
	management_uri = "fd://aziot-edged.mgmt.socket"
	 
	[moby_runtime]
	uri = "unix:///var/run/docker.sock"
	network = "azure-iot-edge"
	EOF

	iotedge config apply -c '/etc/aziot/config.toml'

	apt install deviceupdate-agent 
	apt install deliveryoptimization-plugin-apt
	systemctl restart adu-agent
      ) &
